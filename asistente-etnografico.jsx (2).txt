// ─── API helpers ─────────────────────────────────────────────────────────────

async function streamClaude(system, userContent, onChunk) {
  const res = await fetch("https://api.anthropic.com/v1/messages", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({
      model: "claude-sonnet-4-20250514", max_tokens: 1000, stream: true,
      system,
      messages: [{ role: "user", content: userContent }]
    })
  });
  if (!res.ok) throw new Error(`API ${res.status}`);
  const reader = res.body.getReader();
  const dec = new TextDecoder();
  let buf = "";
  while (true) {
    const { done, value } = await reader.read();
    if (done) break;
    buf += dec.decode(value, { stream: true });
    const lines = buf.split("\n"); buf = lines.pop();
    for (const l of lines) {
      if (!l.startsWith("data: ")) continue;
      const d = l.slice(6).trim();
      if (d === "[DONE]") continue;
      try { const p = JSON.parse(d); if (p.delta?.text) onChunk(p.delta.text); } catch {}
    }
  }
}

async function callDrive(prompt) {
  const res = await fetch("https://api.anthropic.com/v1/messages", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({
      model: "claude-sonnet-4-20250514", max_tokens: 1000,
      messages: [{ role: "user", content: prompt }],
      mcp_servers: [{ type: "url", url: "https://drivemcp.googleapis.com/mcp/v1", name: "google-drive" }]
    })
  });
  const data = await res.json();
  return data.content?.filter(b => b.type === "text").map(b => b.text).join("") || "";
}

// ─── Storage helpers ──────────────────────────────────────────────────────────

async function loadCats() {
  try {
    const r = await window.storage.get("etno-categories");
    return r ? JSON.parse(r.value) : {};
  } catch { return {}; }
}

async function saveCats(newCats) {
  try {
    const existing = await loadCats();
    for (const c of newCats) {
      const k = c.trim().toLowerCase();
      if (k) existing[k] = (existing[k] || 0) + 1;
    }
    await window.storage.set("etno-categories", JSON.stringify(existing));
    return existing;
  } catch { return {}; }
}

async function clearCats() {
  try { await window.storage.delete("etno-categories"); } catch {}
}

function MarkdownText({ text }) {
  const lines = text.split("\n");
  return (
    <div style={{ fontFamily: "'Lora', serif", lineHeight: 1.8, color: "#1C1C14", fontSize: "0.88rem" }}>
      {lines.map((line, i) => {
        const isSection = /^\*\*[^*]+\*\*$/.test(line.trim());
        const isBullet = /^[-•]\s/.test(line);
        const isEmpty = line.trim() === "";

        const render = (t) => {
          const b = t.replace(/\*\*(.*?)\*\*/g, "<strong>$1</strong>");
          const it = b.replace(/\*(.*?)\*/g, "<em>$1</em>");
          return it.replace(/`(.*?)`/g, `<code style="font-family:'IBM Plex Mono',monospace;font-size:0.82em;background:#F0EBE0;padding:1px 5px;border-radius:2px;">$1</code>`);
        };

        if (isSection) return (
          <div key={i} style={{
            fontFamily: "'IBM Plex Mono', monospace", fontWeight: 400,
            fontSize: "0.62rem", color: "#2D6A4F", marginTop: i > 0 ? "1.4em" : 0,
            marginBottom: "0.4em", letterSpacing: "0.14em", textTransform: "uppercase",
            borderBottom: "1px solid #D4C9B0", paddingBottom: "0.25em"
          }}>{line.replace(/\*\*/g, "")}</div>
        );

        if (isBullet) return (
          <div key={i} style={{ display: "flex", gap: "0.6em", marginBottom: "0.3em", paddingLeft: "0.3em" }}>
            <span style={{ color: "#B85C38", marginTop: "0.15em", flexShrink: 0, fontSize: "0.8em" }}>◦</span>
            <span dangerouslySetInnerHTML={{ __html: render(line.replace(/^[-•]\s/, "")) }} />
          </div>
        );

        if (isEmpty) return <div key={i} style={{ height: "0.5em" }} />;

        return <p key={i} style={{ margin: "0 0 0.35em" }} dangerouslySetInnerHTML={{ __html: render(line) }} />;
      })}
    </div>
  );
}

function DrivePanel({ onLoadContent }) {
  const [query, setQuery] = useState("");
  const [files, setFiles] = useState([]);
  const [loading, setLoading] = useState(false);
  const [loadingFile, setLoadingFile] = useState(null);
  const [error, setError] = useState("");
  const [targetMode, setTargetMode] = useState("codificar");

  const search = async () => {
    if (!query.trim()) return;
    setLoading(true); setError(""); setFiles([]);
    try {
      const result = await callDrive(
        `Search Google Drive for files matching: "${query.trim()}". List up to 8 results. Return ONLY a JSON array, no other text, no markdown fences. Format: [{"id":"...","name":"...","type":"doc|sheet|other"}]. If no files found, return [].`
      );
      const m = result.match(/\[[\s\S]*\]/);
      if (m) {
        try { setFiles(JSON.parse(m[0])); }
        catch { setError("No se pudo parsear la respuesta de Drive."); }
      } else {
        setError("No se encontraron archivos. Verificá que tu Google Drive está conectado.");
      }
    } catch (e) {
      setError("Error al conectar con Google Drive. Asegurate de tener el conector activo.");
    }
    setLoading(false);
  };

  const loadFile = async (file) => {
    setLoadingFile(file.id);
    try {
      const content = await callDrive(
        `Get the full text content of the Google Drive file named "${file.name}" (ID: ${file.id}). Return only the document's text content, nothing else. If it's very long, include the first 1500 words.`
      );
      onLoadContent(content, targetMode, file.name);
    } catch {
      setError(`No se pudo cargar "${file.name}".`);
    }
    setLoadingFile(null);
  };

  return (
    <div>
      <div style={{ marginBottom: "1rem" }}>
        <div style={{ fontFamily: "'IBM Plex Mono', monospace", fontSize: "0.6rem", letterSpacing: "0.12em", textTransform: "uppercase", color: "#8A7A68", marginBottom: "0.6rem" }}>
          Cargar en modo
        </div>
        <div style={{ display: "flex", gap: "0.4rem", flexWrap: "wrap", marginBottom: "1.2rem" }}>
          {TOOL_KEYS.map(k => (
            <button key={k} onClick={() => setTargetMode(k)} style={{
              fontFamily: "'Cormorant Garamond', serif", fontSize: "0.85rem",
              padding: "0.3rem 0.75rem", border: "1px solid",
              borderColor: targetMode === k ? "#1E3A2F" : "#D4C9B0",
              background: targetMode === k ? "#1E3A2F" : "transparent",
              color: targetMode === k ? "#F5EDD4" : "#5C5040",
              borderRadius: "3px", cursor: "pointer", transition: "all 0.12s"
            }}>
              {MODES[k].symbol} {MODES[k].label}
            </button>
          ))}
        </div>

        <div style={{ fontFamily: "'IBM Plex Mono', monospace", fontSize: "0.6rem", letterSpacing: "0.12em", textTransform: "uppercase", color: "#8A7A68", marginBottom: "0.4rem" }}>
          Buscar en Drive
        </div>
        <div style={{ display: "flex", gap: "0.5rem" }}>
          <input
            value={query}
            onChange={e => setQuery(e.target.value)}
            onKeyDown={e => e.key === "Enter" && search()}
            placeholder="nombre del documento, carpeta, palabras clave…"
            style={{
              flex: 1, fontFamily: "'IBM Plex Mono', monospace", fontSize: "0.8rem",
              padding: "0.6rem 0.8rem", border: "1px solid #C4B49A",
              borderRadius: "4px", background: "#FEFAF2", color: "#1C1C14",
              outline: "none"
            }}
          />
          <button onClick={search} disabled={loading} style={{
            padding: "0.6rem 1rem", background: "#1E3A2F", color: "#F5EDD4",
            border: "none", borderRadius: "3px", cursor: "pointer",
            fontFamily: "'IBM Plex Mono', monospace", fontSize: "0.7rem",
            letterSpacing: "0.08em", textTransform: "uppercase",
            opacity: loading ? 0.6 : 1
          }}>
            {loading ? "…" : "Buscar"}
          </button>
        </div>
      </div>

      {error && (
        <div style={{ padding: "0.7rem 0.9rem", background: "rgba(184,92,56,0.08)", border: "1px solid rgba(184,92,56,0.25)", borderRadius: "4px", color: "#B85C38", fontFamily: "'IBM Plex Mono', monospace", fontSize: "0.72rem", marginBottom: "1rem" }}>
          {error}
        </div>
      )}

      {files.length > 0 && (
        <div>
          <div style={{ fontFamily: "'IBM Plex Mono', monospace", fontSize: "0.6rem", letterSpacing: "0.12em", textTransform: "uppercase", color: "#8A7A68", marginBottom: "0.6rem" }}>
            {files.length} archivo{files.length !== 1 ? "s" : ""} encontrado{files.length !== 1 ? "s" : ""}
          </div>
          {files.map(f => (
            <div key={f.id} style={{
              padding: "0.8rem 1rem", border: "1px solid #D4C9B0",
              borderRadius: "4px", marginBottom: "0.4rem",
              background: "#FEFAF2", display: "flex",
              alignItems: "center", justifyContent: "space-between",
              gap: "1rem"
            }}>
              <div style={{ display: "flex", alignItems: "center", gap: "0.6rem", overflow: "hidden" }}>
                <span style={{ fontSize: "1rem", flexShrink: 0, opacity: 0.6 }}>
                  {f.type === "doc" ? "📄" : f.type === "sheet" ? "📊" : "📁"}
                </span>
                <span style={{ fontFamily: "'Lora', serif", fontSize: "0.85rem", color: "#1C1C14", overflow: "hidden", textOverflow: "ellipsis", whiteSpace: "nowrap" }}>
                  {f.name}
                </span>
              </div>
              <button
                onClick={() => loadFile(f)}
                disabled={!!loadingFile}
                style={{
                  flexShrink: 0, padding: "0.3rem 0.75rem",
                  fontFamily: "'IBM Plex Mono', monospace", fontSize: "0.6rem",
                  letterSpacing: "0.1em", textTransform: "uppercase",
                  border: "1px solid #2D6A4F", color: "#2D6A4F",
                  background: "transparent", borderRadius: "3px",
                  cursor: "pointer", transition: "all 0.12s",
                  opacity: loadingFile === f.id ? 0.5 : 1
                }}
              >
                {loadingFile === f.id ? "cargando…" : "→ cargar"}
              </button>
            </div>
          ))}
        </div>
      )}

      {!loading && files.length === 0 && !error && (
        <div style={{ textAlign: "center", padding: "2.5rem 1rem", color: "#C4B49A" }}>
          <div style={{ fontSize: "2rem", marginBottom: "0.5rem", opacity: 0.35 }}>☁</div>
          <div style={{ fontFamily: "'Cormorant Garamond', serif", fontStyle: "italic", fontSize: "0.9rem" }}>
            Buscá notas de campo, entrevistas<br />o documentos en tu Google Drive
          </div>
        </div>
      )}
    </div>
  );

export default function AsistenteEtnografico() {
  const [mode, setMode] = useState("codificar");
  const [input, setInput] = useState("");
  const [output, setOutput] = useState("");
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState("");
  const [toast, setToast] = useState("");
  const [cats, setCats] = useState({});
  const [history, setHistory] = useState([]);
  const outputRef = useRef(null);

  useEffect(() => { loadCats().then(setCats); }, []);

  useEffect(() => {
    if (outputRef.current) outputRef.current.scrollTop = outputRef.current.scrollHeight;
  }, [output]);

  const showToast = (msg) => {
    setToast(msg);
    setTimeout(() => setToast(""), 3000);
  };

  const handleSubmit = async () => {
    if (!input.trim() || loading) return;
    setLoading(true); setError(""); setOutput("");
    let acc = "";
    try {
      const m = MODES[mode];
      await streamClaude(m.system, input.trim(), chunk => {
        acc += chunk;
        setOutput(acc);
      });

      // Extract and save categories if codificar mode
      if (mode === "codificar") {
        const catsMatch = acc.match(/\[CATS\]:\s*(.+)/);
        if (catsMatch) {
          const newCats = catsMatch[1].split("|").map(c => c.trim()).filter(Boolean);
          const updated = await saveCats(newCats);
          setCats(updated);
          showToast(`${newCats.length} categoría${newCats.length !== 1 ? "s" : ""} guardada${newCats.length !== 1 ? "s" : ""} ✓`);
          acc = acc.replace(/\n?\[CATS\]:.*$/, "").trim();
          setOutput(acc);
        }
      }

      setHistory(prev => [{
        mode, label: MODES[mode].label, symbol: MODES[mode].symbol,
        input: input.trim(), output: acc,
        time: new Date().toLocaleTimeString("es-AR", { hour: "2-digit", minute: "2-digit" })
